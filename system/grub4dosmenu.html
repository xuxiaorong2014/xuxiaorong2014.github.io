配置文件 menu.lst

color white/blue yellow/red white/blue white/blue
timeout 60
color border=0xEEFFEE
default /default
debug off
font /boot/grub/font_cn.gz
graphicsmode -1 640:800 320:600 16:24 || graphicsmode -1 -1 -1 24:32
splashimage /boot/grub/back.lzma
setmenu --font-spacing=1:8 --string=20=1=0xFFFFFF="开票机系统选择菜单"
setmenu --box x=8 w=40 y=3 h=8 l=1 --ver-off
setmenu --lang=zh --auto-num-on
